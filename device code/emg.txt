#include <Adafruit_NeoPixel.h>

#define EMG_PIN   A0
#define LED_PIN   6
#define LED_COUNT 30
#define BAUD_RATE 115200

Adafruit_NeoPixel strip(LED_COUNT, LED_PIN, NEO_GRB + NEO_KHZ800);

int baseline = 0;

int clampi(int x, int lo, int hi) {
  if (x < lo) return lo;
  if (x > hi) return hi;
  return x;
}

uint32_t colorByAmp(int amp) {
  // 最小幅度绿色，幅度越大颜色越“热”
  if (amp <= 5)        return strip.Color(0, 180, 0);     // 绿（最小）
  else if (amp <= 12)  return strip.Color(120, 180, 0);   // 黄绿
  else if (amp <= 20)  return strip.Color(180, 120, 0);   // 黄
  else if (amp <= 30)  return strip.Color(220, 60, 0);    // 橙
  else if (amp <= 45)  return strip.Color(220, 0, 0);     // 红
  else                 return strip.Color(160, 0, 220);   // 紫（最强）
}

void setup() {
  Serial.begin(BAUD_RATE);

  strip.begin();
  strip.show(); // 全灭

  // 采基线：上电后请放松 2 秒
  long sum = 0;
  const int N = 500;
  for (int i = 0; i < N; i++) {
    sum += analogRead(EMG_PIN);
    delay(4); // 约2秒
  }
  baseline = sum / N;

  Serial.print("baseline=");
  Serial.println(baseline);
}

void loop() {
  // 平滑读数
  static float smooth = 0;
  int v = analogRead(EMG_PIN);
  smooth = 0.85f * smooth + 0.15f * v;

  // 强度：偏离基线的幅度
  int amp = abs((int)smooth - baseline);

  // 把 amp 映射成亮灯数量（你可调上限 60）
  int level = map(clampi(amp, 0, 60), 0, 60, 0, LED_COUNT);

  // 颜色随 amp 变化
  uint32_t c = colorByAmp(amp);

  // 点亮前 level 颗
  strip.clear();
  for (int i = 0; i < level; i++) {
    strip.setPixelColor(i, c);
  }
  strip.show();

  // 串口输出：方便你看 amp 和 level
  Serial.print(v);
  Serial.print(", smooth=");
  Serial.print((int)smooth);
  Serial.print(", amp=");
  Serial.print(amp);
  Serial.print(", level=");
  Serial.println(level);

  delay(20);
}
